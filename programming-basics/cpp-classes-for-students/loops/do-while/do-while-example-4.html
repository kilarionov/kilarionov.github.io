РЕШЕНИЕ № 1 на пример № 4 за DO-WHILE

// Да се отпечатат в обратен ред цифрите на n
#include <iostream>
using namespace std ;
main ()
{
  int n, lastDigit ;
  cin >>n ;
  n = (n < 0) ? -n : n ;

  do
  {
    lastDigit = n % 10 ;
    cout <<lastDigit ;
  }
   while (n = n/10) ;
}
