DO-WHILE EXAMPLE #5

РЕШЕНИЕ на пример № 5 за DO-WHILE

// Да се отпечата сумата от цифрите на n
#include <iostream>
using namespace std ;
main ()
{
  int n, lastDigit, sum=0 ;
  cout <<"n=" ; cin >>n ;
  n = (n < 0) ? -n : n ;
  
  do
    sum += lastDigit = n % 10 ;
  while (n = n/10) ;
  
  cout <<"\tsum=" <<sum <<endl ;
}
